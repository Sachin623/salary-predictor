<script>
let chart;

function predictSalary() {
    // Get inputs
    let experience = parseInt(document.getElementById("experience").value);
    let education = document.getElementById("education").value;
    let job = document.getElementById("job").value;

    // Salary calculation (mock formula)
    let base = 20000;
    let expFactor = experience * 3000;
    let eduFactor = (education === "Masters") ? 10000 : (education === "PhD") ? 20000 : 0;
    let jobFactor = (job === "Project Manager") ? 15000 : (job === "Data Scientist") ? 12000 : 8000;

    let predictedSalary = base + expFactor + eduFactor + jobFactor;

    // Update salary output
    document.getElementById("salaryOutput").textContent = "Predicted Salary: ₹ " + predictedSalary.toLocaleString();

    // Update Chart
    updateChart(experience, predictedSalary);
}

function updateChart(exp, salary) {
    const ctx = document.getElementById('salaryChart').getContext('2d');

    if (chart) {
        chart.destroy();
    }

    chart = new Chart(ctx, {
        type: 'bar',
        data: {
            labels: [exp + " Years"],
            datasets: [{
                label: 'Salary',
                data: [salary],
                backgroundColor: 'rgba(255, 99, 132, 0.7)'
            }]
        },
        options: {
            scales: {
                y: { beginAtZero: true }
            }
        }
    });
}

// Initial chart load
updateChart(8, 76656);
</script>
